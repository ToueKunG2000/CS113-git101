#include <stdio.h> คือการเรียกใช้ Library ที่รวบรวมฟังก์ชันที่สามารถใช้งานได้ใน stdio.h
#define VARIABLE VALUE คือการกำหนดค่าคงที่ จะสามารถเปลี่ยนได้จากตรงนี้เท่านั้น
การประกาศค่าตัวแปร
datatype variable = initial;
unsigned = ใส่หน้า datatype เพื่อระบุว่าเก็บค่าที่เป็นบวกเท่านั้น
char = ประเภทข้อมูลของตัวอักษร 1 ตัว ขนาดข้อมูล 1 byte
short = ประเภทข้อมูลของค่าจำนวนเต็ม ขนาดข้อมูล 2 bytes
int = ประเภทข้อมูลของค่าจำนวนเต็ม ขนาดข้อมูล 4 bytes
long = ประเภทข้อมูลของค่าจำนวนเต็ม ขนาดข้อมูล 8 bytes
float = ประเภทข้อมูลของค่าจำนวนจริง เก็บค่าทศนิยมที่ถูกต้องได้ 7 หลัก ขนาดข้อมูล 4 bytes
double = ประเภทข้อมูลของค่าจำนวนจริง เก็บค่าทศนิยมที่ถูกต้องได้ 15 หลัก ขนาดข้อมูล 8 bytes
printf("format",variable); คือการเรียกใช้ฟังก์ชัน printf ซึ่งอยู่ใน Library : stdio.h

ใน format เราสามารถใส่ตัวอักษร และ ค่า format ได้ เช่น 
printf("%d",num); การให้แสดงค่าของ num ในรูปจำนวนเต็ม
printf("%c",char); การแสดงค่าตัวอักษร char
printf("%f",float); การแสดงค่าของ float ในรูปจำนวนจริง
